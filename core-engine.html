// context.ts — 系统上下文构建
export const getSystemContext = memoize(async () => {
  const gitStatus = await getGitStatus();
  return { ...(gitStatus && { gitStatus }) };
});

export const getUserContext = memoize(async () => {
  const claudeMd = getClaudeMds(
    filterInjectedMemoryFiles(await getMemoryFiles())
  );
  return {
    ...(claudeMd && { claudeMd }),
    currentDate: `Today's date is ${getLocalISODate()}.`,
  };
});
💡 设计洞察：系统上下文和用户上下文都使用 lodash.memoize 缓存——在整个会话期间只计算一次。Git 状态是启动时的快照，不会在对话中实时更新。这是有意为之：避免长会话中 Git 状态变化导致 System Prompt 变动、打破 Prompt Cache。

buildTool() 工厂函数

所有工具通过 buildTool() 创建，它提供安全的默认值（fail-closed 原则）：

▶ buildTool() 默认值与设计思想
const TOOL_DEFAULTS = {
  isEnabled: () => true,
  isConcurrencySafe: () => false,  // 默认：不安全（保守）
  isReadOnly: () => false,          // 默认：可能写入
  isDestructive: () => false,
  checkPermissions: (input) =>      // 默认：允许（委托通用权限）
    Promise.resolve({ behavior: 'allow', updatedInput: input }),
  toAutoClassifierInput: () => '',  // 默认：跳过安全分类
  userFacingName: () => '',
};

export function buildTool<D extends AnyToolDef>(def: D): BuiltTool<D> {
  return { ...TOOL_DEFAULTS, userFacingName: () => def.name, ...def };
}

sed 编辑的安全处理

对 sed -i 命令有特殊处理——不直接执行，而是模拟执行后展示 diff：

1 parseSedEditCommand() 解析 sed 表达式
↓
2 内存中模拟执行，生成 diff 预览
↓
3 弹出权限对话框展示文件变更预览
↓
4 确认后 applySedEdit() 直接写文件（不调用 sed）
🔑 设计思想：遵循 fail-closed 原则——无法解析的命令一律当作危险命令。管道命令需要每个子命令都只读才判定整体只读。sleep > 2s 被拦截防止轮询滥用。

压缩之间的协调关系

  • task_budget：跨压缩边界追踪 taskBudgetRemaining（压缩后服务端看不到完整历史）

StreamingToolExecutor 并行执行

  • 当 streamingToolExecution gate 开启时，工具 在 API streaming 过程中就开始执行
  • 每当流中出现完整的 tool_use block → 立即提交给 executor
  • Fallback 发生时 → executor.discard() 丢弃孤儿结果 + 创建新实例
  • Tombstone 机制：失败的 partial thinking blocks 被标记为 tombstone → 从 UI 和 transcript 中移除

38.3 迁移调度器

runMigrations() 在启动时检查 migrationVersion（当前 v11）， 按序执行 11 个幂等迁移。版本号只在所有迁移成功后更新—— crash-safe（崩溃后下次启动会重新执行未完成的迁移）。

38.4 延迟预热

startDeferredPrefetches() 在首次渲染 之后 才执行—— 利用「用户正在打字」的时间窗口预热缓存：

  • initUser() — 用户信息
  • getUserContext() — CLAUDE.md 记忆加载
  • countFilesRoundedRg() — 项目规模估算（3s 超时）
  • refreshModelCapabilities() — 模型能力矩阵
  • settingsChangeDetector / skillChangeDetector — 文件监听器
  • Ant-only: eventLoopStallDetector — 主线程阻塞检测（>500ms 报警）
🔑 性能工程：main.tsx 的设计目标是「从敲下 claude 到看到 REPL」控制在 200ms 内。通过 import 阶段并行预热（MDM + Keychain）、延迟加载非核心模块（coordinator/assistant 条件 require）、延迟预热（首屏后才启动缓存预加载），实现了极致的启动性能。
