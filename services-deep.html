// paramsFromContext 核心逻辑（简化）
const paramsFromContext = (retryContext: RetryContext) => {
  // 1. Beta headers 动态追加（1M / Bedrock / ToolSearch）
  const betasParams = [...betas]
  if (getSonnet1mExpTreatmentEnabled(retryContext.model))
    betasParams.push(CONTEXT_1M_BETA_HEADER)

  // 2. Effort + TaskBudget → output_config
  const outputConfig: BetaOutputConfig = {}
  configureEffortParams(effort, outputConfig, extraBody, betasParams, model)
  configureTaskBudgetParams(taskBudget, outputConfig, betasParams)

  // 3. Thinking: adaptive vs budget
  if (hasThinking && modelSupportsAdaptiveThinking(model)) {
    thinking = { type: 'adaptive' }
  } else {
    thinking = { budget_tokens: min(maxOutput-1, budget), type: 'enabled' }
  }

  // 4. Context Management API（服务端工具清理）
  const contextManagement = getAPIContextManagement({
    hasThinking, isRedactThinkingActive, clearAllThinking: thinkingClearLatched
  })

  // 5. Fast Mode: header latched(cache-safe), speed dynamic(cooldown)
  if (isFastModeForRetry) speed = 'fast'

  // 6. Cache breakpoints + global scope
  messages: addCacheBreakpoints(messagesForAPI, enablePromptCaching, ...)
}

📊 Reactive Compact 集成 — Prompt Too Long 智能处理
Prompt Too Long 处理流程：

1. 解析 API 响应中的 max_tokens / requested_tokens
2. 计算 token gap = requested - max
3. 通过 getPromptTooLongTokenGap 存储 gap
4. 下一次 compact 时：
   ├── 如果 gap > 单 group 大小 → 一次跳过多个 group
   ├── 而非逐 group 压缩（省去多轮 API 调用）
   └── 快速收敛到可用上下文窗口大小

🔐 OAuth 认证链 — claude.ai Proxy Fetch
// createClaudeAiProxyFetch — claude.ai MCP 代理认证
// 关键设计：401 自动重试 + 并发安全
export function createClaudeAiProxyFetch(innerFetch: FetchLike): FetchLike {
  return async (url, init) => {
    const doRequest = async () => {
      await checkAndRefreshOAuthTokenIfNeeded()
      const currentTokens = getClaudeAIOAuthTokens()
      headers.set('Authorization', `Bearer ${currentTokens.accessToken}`)
      const response = await innerFetch(url, { ...init, headers })
      // 返回发送时的 token（非最新 token）防止并发 refresh 竞态
      return { response, sentToken: currentTokens.accessToken }
    }
    const { response, sentToken } = await doRequest()
    if (response.status !== 401) return response
    // handleOAuth401Error 返回 true → token 已更换 → 重试
    const tokenChanged = await handleOAuth401Error(sentToken)
    if (!tokenChanged) {
      // ELOCKED 竞态：另一个连接器可能已 refresh
      const now = getClaudeAIOAuthTokens()?.accessToken
      if (!now || now === sentToken) return response
    }
    return (await doRequest()).response
  }
}

💭 autoDream — 后台记忆整合引擎（550行） 35

autoDream/（4 文件/550行）实现了 Claude Code 的 「做梦」机制—— 不是比喻，而是字面意义上的：当用户不使用 Claude 时（空闲 ≥24 小时）， 系统自动启动一个子 Agent 审查最近的会话记录，将分散的知识整合写入记忆文件。

35.1 三层门控（Gate）

门控条件开销目的
时间门距上次整合 ≥ minHours（默认 24h）1 次 stat 调用防止频繁触发
会话门新会话数 ≥ minSessions（默认 5）目录扫描（10分钟节流）确保有足够素材
锁门tryAcquireConsolidationLock() 成功文件锁防并发整合

35.2 执行流程

  • 触发点：每次 query 响应后的 stopHooks（postSamplingHooks）
  • 子 Agent 执行：通过 runForkedAgent() 启动隔离执行环境，使用 consolidationPrompt
  • 工具约束：Bash 限制为只读命令（ls/find/grep/cat），File Edit/Write 只允许操作 memoryRoot
  • 进度监控：makeDreamProgressWatcher() 跟踪每个 assistant turn 的内容和工具调用
  • 完成回调：通过 appendSystemMessage 在主对话中插入 "Improved N memory files" 通知
  • 失败回滚：rollbackConsolidationLock(priorMtime) 回退时间戳，让下次重试

35.3 与 Kairos 的关系

当 getKairosActive() 为 true 时，autoDream 自动禁用——因为 Kairos 模式下 有自己的 disk-skill dream 机制。Remote 模式下同样禁用（远程环境的文件系统权限不同）。 配置来自 GrowthBook A/B 测试系统 (tengu_onyx_plover feature flag)。

💡 PromptSuggestion — 智能建议引擎（1514行） 35.4

PromptSuggestion/（2 文件/1514行）在用户输入时提供上下文感知的命令/操作建议。 它分析当前项目结构、对话历史、工具使用模式，生成个性化的"接下来你可能想做..."提示。 建议来源包括：最近的 git 变更、未完成的 TODO、常用工具组合模式、 以及基于 GrowthBook 的 A/B 实验推荐。建议结果缓存到内存中，每次用户 idle 时刷新。

📚 SessionMemory — 会话记忆持久化（1026行） 35.5

SessionMemory/（3 文件/1026行）管理跨会话的记忆持久化。每次会话结束时， 自动提取关键知识点（代码架构决策、用户偏好、项目约束）写入结构化记忆文件。 记忆按 scope 分层存储——全局 (~/.claude/memory/)、项目 (.claude/memory/)。 加载时按优先级合并：项目级覆盖全局级。与 extractMemories 服务（769行）协作， 后者负责从对话消息中自动提取值得记忆的知识片段。

👥 teamMemorySync — 团队记忆同步（2167行） 35.6

teamMemorySync/（5 文件/2167行）实现团队级记忆的同步与冲突解决。当团队成员共享 同一项目的 .claude/ 目录时，记忆文件可能产生冲突。该服务使用 CRDT-like 的合并策略——每条记忆条目带有时间戳和作者标识，冲突时按"最新写入者优先"合并， 并保留冲突详情供人工审核。同步通过 git 机制触发（push/pull 时自动检测变更）。
