18.1 会话持久化引擎

sessionStorage.ts（5105行）实现了 Claude Code 的会话事务日志——每条消息、每个工具调用、每个状态变更都以 JSONL 格式追加写入磁盘。这是会话恢复（--resume）和子 Agent 管理的基础。

▶ 写队列微批机制 sessionStorage.ts:530-686
class Project {
  private writeQueues = new Map<string, Array<{ entry: Entry; resolve: () => void }>>();
  private FLUSH_INTERVAL_MS = 100;       // 100ms 定时排空
  private readonly MAX_CHUNK_BYTES = 100 * 1024 * 1024; // 100MB 分块

  // 写入不是立即执行——而是入队
  private enqueueWrite(filePath: string, entry: Entry): Promise<void> {
    return new Promise<void>(resolve => {
      let queue = this.writeQueues.get(filePath);
      if (!queue) { queue = []; this.writeQueues.set(filePath, queue); }
      queue.push({ entry, resolve });
      this.scheduleDrain();  // ← 100ms 延迟触发
    });
  }

  // 排空时批量拼接 JSONL + 分块写入
  private async drainWriteQueue(): Promise<void> {
    for (const [filePath, queue] of this.writeQueues) {
      const batch = queue.splice(0);
      let content = '';
      for (const { entry, resolve } of batch) {
        const line = jsonStringify(entry) + '\n';
        if (content.length + line.length >= this.MAX_CHUNK_BYTES) {
          await this.appendToFile(filePath, content); // 分块写入
          content = '';
        }
        content += line;
      }
      if (content.length > 0) await this.appendToFile(filePath, content);
    }
  }
}
// 效果：流式传输时 ~10 个 entries/100ms 合并为一次 I/O
// 而非每个 token 到达都触发一次文件追加

▶ UUID 派生算法 messages.ts:722-728
// 问题：一个 API 响应包含多个 content block（text + tool_use + thinking）
// 规范化时拆成多条消息，每条需要唯一 UUID
// 但不能用 randomUUID()——否则相同输入产生不同输出，破坏 memo 缓存

// 解决：确定性派生
export function deriveUUID(parentUUID: UUID, index: number): UUID {
  const hex = index.toString(16).padStart(12, '0');
  return `${parentUUID.slice(0, 24)}${hex}` as UUID;
}
// 取父 UUID 前 24 字符 + index 的 12 位 hex 后缀
// 结果：同一输入永远产生相同 UUID 组
// React 的 key 稳定，memo 不会被击穿

// 短 ID 派生（用于 snip tool 引用）：
export function deriveShortMessageId(uuid: string): string {
  const hex = uuid.replace(/-/g, '').slice(0, 10);
  return parseInt(hex, 16).toString(36).slice(0, 6); // 6字符 base36
}

🔑 一句话总结：utils/ 不是“工具函数集”——它是 Claude Code 的平台层。会话持久化引擎确保每次对话可恢复；消息类型系统定义了 API 数据契约；Hook 引擎提供了用户可编程性；配置系统管理 500+ 个的状态字段；认证层支持三种提供商。这些模块合在一起，构成了一个可以挂载任意功能的基础设施。
